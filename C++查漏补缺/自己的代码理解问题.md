#### 指针作为函数参数
```
void fun(int* p)
{
	int b = 100;
    p = &b;
}
int main()
{
	int a = 10;
    int* p1 = &a;
    cout<< p1 <<endl;
    fun(p1);  //给函数传一个指针，接收也是一个指针，其实传的是这个指针的拷贝，在函数里改变指针内存，并不能改变原来的。
    cout<< p1<< endl;
	return 0;
}
```
解决
```
void fun(int** p)
{
	int b = 100;
    p = &b;
}
int main()
{
	int a = 10;
    int* p1 = &a;
    cout<< p1 <<endl;
    fun(&p1);  //给函数传一个指针的本身的地址，函数处用一个指向指针地址的指针接收
    cout<< p1<< endl;
	return 0;
}
```
